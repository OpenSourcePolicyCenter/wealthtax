\title{Firms Problem with CES Production and Ownership of Capital}
\maketitle

Firm $i$'s maximization problem is:
\begin{equation}\label{EqProfMax}
  \begin{split}
   V_i(k_{it},\Omega_t) & = \max_{k_{i,t+1},n_{it}} \left[ p_{it} f(z_t,k_{it},n_{it}) - w_tn_{it} + (1-\delta)k_{it} - k_{i,t+1} \right] + \frac{1}{1+r_{t+1}} V_i(k_{i,t+1},\Omega_{t+1});\\
   & f(z_t,k_{it},n_{it}) = z_t \left[ \alpha k_{it}^\eta + (1-\alpha)n_{it}^\eta \right]^{\frac{1}{\eta}}
  \end{split}
\end{equation}
